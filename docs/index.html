def inttoday(i):
    if ((datetime.datetime.now().weekday() + i)%7 ==0):
        return 'Lundi'
    elif ((datetime.datetime.now().weekday()+i)%7 ==1):
        return 'Mardi'
    elif ((datetime.datetime.now().weekday()+i)%7 ==2):
